#!/usr/bin/env python3

import cv2
import numpy as np


def main():
    cap = cv2.VideoCapture(0)
    if not cap.isOpened():
        return

    kernel = np.ones((5, 5), np.uint8)

    while True:
        ret, frame = cap.read()
        if not ret: break

        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)

        # --- Masks ---
        # Red (combined)
        r_low1 = np.array([0, 120, 100]);
        r_high1 = np.array([10, 255, 255])
        r_low2 = np.array([170, 120, 100]);
        r_high2 = np.array([180, 255, 255])
        red_mask = cv2.inRange(hsv, r_low1, r_high1) | cv2.inRange(hsv, r_low2, r_high2)

        # Black
        black_mask = cv2.inRange(hsv, np.array([0, 0, 0]), np.array([180, 255, 75]))
        black_mask = cv2.morphologyEx(black_mask, cv2.MORPH_OPEN, kernel)

        # Data structures to hold coordinates: (x, y, w, h, centerX, centerY)
        red_objs = []
        black_objs = []

        # Find Red Objects
        contours_r, _ = cv2.findContours(red_mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        for c in contours_r:
            if cv2.contourArea(c) > 1000:
                x, y, w, h = cv2.boundingRect(c)
                red_objs.append((x, y, w, h, x + w // 2, y + h // 2))
                cv2.rectangle(frame, (x, y), (x + w, y + h), (0, 0, 255), 2)

        # Find Black Objects
        contours_b, _ = cv2.findContours(black_mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        for c in contours_b:
            if cv2.contourArea(c) > 1000:
                x, y, w, h = cv2.boundingRect(c)
                black_objs.append((x, y, w, h, x + w // 2, y + h // 2))
                cv2.rectangle(frame, (x, y), (x + w, y + h), (255, 255, 255), 2)

        # ===== SANDWICH DETECTION LOGIC =====
        pattern_detected = False

        for b in black_objs:
            bx, by = b[4], b[5]  # Black Center

            for i in range(len(red_objs)):
                for j in range(i + 1, len(red_objs)):
                    r1x, r1y = red_objs[i][4], red_objs[i][5]
                    r2x, r2y = red_objs[j][4], red_objs[j][5]

                    # Check Horizontal Sandwich (Left - Black - Right)
                    horiz = (min(r1x, r2x) < bx < max(r1x, r2x)) and (abs(r1y - r2y) < 100)

                    # Check Vertical Sandwich (Up - Black - Down)
                    vert = (min(r1y, r2y) < by < max(r1y, r2y)) and (abs(r1x - r2x) < 100)

                    if horiz or vert:
                        pattern_detected = True
                        # Highlight the middle object
                        cv2.circle(frame, (bx, by), 20, (0, 255, 255), -1)
                        cv2.putText(frame, "RED-BLACK-RED PATTERN!", (50, 50),
                                    cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 255), 3)

        if pattern_detected:
            print("RED AND BLACK PATTERN DETECTED")

        cv2.imshow('Detection', frame)
        if cv2.waitKey(1) & 0xFF == ord('q'): break

    cap.release()
    cv2.destroyAllWindows()


if __name__ == '__main__':
    main()
